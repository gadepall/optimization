\paragraph{\begin{flushleft}\textbf{Problem: }
\textbf{Minimise}
\begin{align*}
Z = -3x+4y 
\end{align*}
subjected to
\begin{align*}
x+2y \preceq 8, 3x+2y \preceq 12, x \succeq 0 , y \succeq 0
\end{align*}
\includegraphics[scale=0.5]{fig.pdf} 
\end{flushleft}}
\section*{Solution}
\begin{flushleft}
Problem can be formulated as,\\
\end{flushleft}
\begin{align}
\min_{\vec{x}} Z=(-3x+4y)
\end{align}
\begin{align}
x+2y \preceq8
\end{align}
\begin{align}
3x+2y \preceq 12
\end{align}
\begin{align}
x\succeq0,y\succeq0
\end{align}
eq 2 and 3 to 4 can be expressed in vector form as
\begin{align*}
\min_{\vec{x}}\vec{Z}=\myvec{-3 \hspace{0.2cm}4}\vec{x}\\
\myvec{1 \hspace{0.2cm} 2\\
       3 \hspace{0.2cm} 2\\
       1 \hspace{0.2cm}0\\
       0 \hspace{0.2cm} 1}\vec{x}\succeq \myvec{8 \\12\\0\\0}
\end{align*}
Solving above equations using cvxpy, we get\\
\vspace{0.1cm}\\
\begin{align}
\min_{\vec{x}} Z=-12
\end{align}
\begin{align}
\vec{x}=\myvec{4\\0}
\end{align}
